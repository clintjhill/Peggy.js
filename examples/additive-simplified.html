Additive - Simplified Sample

Source

var sample = new Peggy('sample');

sample.root('additive', sample.sequence(':number', ':plus', ':number'), function(value){
	return value['number'][0] + value['number'][1];
});

sample.rule('number', sample.sequence(/\d+/, /\s+/), function(value){
	return new Number(value[0]);
});

sample.rule('plus', sample.sequence('+', /\s+/));
